CREATE TABLE ingredients (
    ingredient_id INT, 
    ingredient_name CHAR(100), 
    recipe_id INT, 
    FOREIGN KEY (recipe_id) REFERENCES recipes(recipe_id), 
    measurement CHAR(50)
);  

CREATE TABLE recipes (
    recipe_id INT, 
    recipe_name CHAR(100), 
    instructions_id INT
);


CREATE TABLE recipe_steps (
    step_num INT, 
    recipe_id INT, 
    FOREIGN KEY (recipe_id) REFERENCES recipes(recipe_id), 
    step_description TEXT
);
